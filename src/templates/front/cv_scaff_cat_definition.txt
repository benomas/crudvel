import CvResourceMap  from 'src/customs/crudvuel/resource/CvResourceMap'
import OwnService     from 'src/services/<##[singular|studly](resource)##>'
import filler         from 'src/fillers/<##[singular|camel](resource)##>'

let resourceName = '<##[plural|slug](resource)##>'

export default function (store) {
  let resource = new CvResourceMap({
    store,
    name             : resourceName,
    icon             : 'fas fa-list',
    path             : resourceName,
    keyName          : 'id',
    pluralName       : '<##[plural|slug](resource)##>',
    singularName     : '<##[singular|slug](resource)##>',
    crudServices     : new OwnService(store,resourceName),
    resourceSections : ['catalogs'],
    filler           : () => filler(store)
  })

  return resource.setActions([
    {
      name       : 'index',
      icon       : '',
      getService : resource.loadService('index'),
      resource   : resource,
      path       : `${resource.path}`,
      component  : () => import('src/components/resources/<##[plural|slug](resource)##>/CvIndex'),
      type       : 'rows'
    },

    {
      name       : 'create',
      icon       : '',
      setService : resource.loadService('store'),
      resource   : resource,
      path       : `${resource.path}/create`,
      component  : () => import('src/components/resources/<##[plural|slug](resource)##>/CvCreate')
    },

    {
      name          : 'show',
      icon          : '',
      getService    : resource.loadService('show'),
      resource      : resource,
      path          : `${resource.path}/:id`,
      urlParams     : ['id'],
      component     : () => import('src/components/resources/<##[plural|slug](resource)##>/CvShow'),
      disableFields : true,
      isParentRoute : true
    },

    {
      name       : 'edit',
      icon       : '',
      getService : resource.loadService('show'),
      setService : resource.loadService('update'),
      resource   : resource,
      path       : `${resource.path}/:id/edit`,
      urlParams  : ['id'],
      component  : () => import('src/components/resources/<##[plural|slug](resource)##>/CvEdit')
    },

    {
      name       : 'delete',
      icon       : '',
      getService : resource.loadService('show'),
      setService : resource.loadService('destroy'),
      resource   : resource,
      path       : `${resource.path}/:id/delete`,
      urlParams  : ['id'],
      component  : null
    },

    {
      name       : 'activate',
      icon       : '',
      setService : resource.loadService('activate'),
      resource   : resource,
      path       : `${resource.path}/:id/activate`,
      urlParams  : ['id'],
      component  : null
    },

    {
      name       : 'deactivate',
      icon       : '',
      setService : resource.loadService('deactivate'),
      resource   : resource,
      path       : `${resource.path}/:id/deactivate`,
      urlParams  : ['id'],
      component  : null
    },

    {
      name       : 'exportings',
      icon       : '',
      getService : resource.crudServices.exportings,
      resource   : resource,
      path       : `${resource.path}/:id/exportings`,
      urlParams  : ['id'],
      component  : null
    }/*,

    {
      name       : 'import',
      icon       : 'fas fa-file-import',
      setService : resource.loadService('import'),
      resource   : resource,
      path       : `${resource.path}/import`,
      component  : () => import('src/components/resources/<##[plural|slug](resource)##>/CvImport'),
      type       : 'rows'
    }*/
  ]).resourcelangLoader()
}
